8  The below has been updated to work robustly in later environments

10  directories. The macro will NOT create a new ROOT folder - not
11  because it can't, but more because that is generally not something
12  your administrator would like you to do!
13 
14  The macro is idempotent - if you run it twice, it will only create a folder
15  once.

31 %macro mm_createfolder(path=,mDebug=0,outds=mm_createfolder);
32 %local dbg;
33 %if &mDebug=0 %then %let dbg=*;
34 
35 data &outds(keep=folderPath folderuri);
36  length objId parentFolderObjId objType PathComponent rootpath $200
37  folderPath walkpath folderuri $1000;
38  call missing (of _all_);
39  folderPath = cats(symget('path'));
40 
41  * remove any trailing slash ;
42  if ( substr(folderPath,length(folderPath),1) = '/' ) then
43  folderPath=substr(folderPath,1,length(folderPath)-1);
44 
45  * name must not be blank;
46  if ( folderPath = '' ) then do;
47  put "%str(ERR)OR: &sysmacroname PATH parameter value must be non-blank";
48  stop;
49  end;
50 
51  * must have a starting slash ;
52  if ( substr(folderPath,1,1) ne '/' ) then do;
53  put "%str(ERR)OR: &sysmacroname PATH parameter value must have starting slash";
54  stop;
55  end;
56 
57  * check if folder already exists ;
58  rc=metadata_pathobj('',cats(folderPath,"(Folder)"),"",objType,objId);
59  if rc ge 1 then do;
60  put "NOTE: Folder " folderPath " already exists!";
61  stop;
62  end;
63 
64  * do not create a root folder ;
65  if countc(folderPath,'/')=1 then do;
66  put "%str(ERR)OR: &sysmacroname will not create a new ROOT folder";
67  stop;
68  end;
69 
70  rootpath='/'!!scan(folderPath,1,'/');
71  rc=metadata_pathobj('',cats(rootpath,"(Folder)"),"",objType,objId);
72  if rc<1 then do;
73  put "%str(ERR)OR: &sysmacroname: root folder does not exist! " rootpath;
74  put (_all_)(=);
75  stop;
76  end;
77 
78  * walk the path creating (or ignoring) each folder in turn;
79  pathIndex=2;
80  parentFolderObjId=objId;
81  walkpath=rootpath;
82  do until (0);
83  * get the next path component in the folder tree path
84  (using forward slash delimiters);
85  PathComponent = scan(folderPath, pathIndex, '/');
86 
87  * when the path component is blank we have reached the end;
88  if ( PathComponent eq '' ) then leave;
89 
90  put 'NOTE: Looking for path tree folder: ' PathComponent;
91  walkpath=cats(walkpath,'/',PathComponent);
92  objid='';
93  rc=metadata_pathobj('',cats(walkpath,"(Folder)"),"",objType,objId);
94 
95  if rc =1 then do;
96  put 'NOTE- Found: ' walkpath ' uri: ' objid;
97  parentFolderObjId=objid;
98  goto loop_next;
99  end;
100  else if ( rc > 1 ) then do;
101  put "%str(ERR)OR: multiple matching metadata objects found for tree folder: " walkpath;
102  stop;
103  end;
104 
105  put / 'NOTE- Attempting to create metadata folder: ' walkpath;
106  * now we can do the easy bit - creating the tree folder iself;
107  folderuri='';
108  rc = metadata_newobj('Tree', folderuri, PathComponent, '', parentFolderObjId, 'SubTrees');
109  put 'NOTE- metadata_newobj: ' rc= folderuri= parentFolderObjId= ;
110  if ( rc = -1 ) then do;
111  put "%str(ERR)OR: failed to connect to the metadata server";
112  stop;
113  end;
114  else if ( rc = -2 ) then do;
115  put "%str(ERR)OR: failed to create new metadata Tree object for folder: " walkpath;
116  stop;
117  end;
118  else if ( rc ne 0 ) then do;
119  put "%str(ERR)OR: unknown error creating new metadata Tree object for folder: " walkpath;
120  stop;
121  end;
122 
123  * tag the new tree folder with the attribute TreeType=BIP Folder;
124  rc = metadata_setattr(folderuri, 'TreeType', 'BIP Folder');
125  &dbg put 'NOTE- metadata_setattr (TreeType): ' rc= ;
126  if ( rc ne 0 ) then do;
127  put "%str(ERR)OR: failed to set TreeType attribute for new folder: " walkpath;
128  stop;
129  end;
130 
131  * tag the new tree folder with the SAS 9.2 attribute PublicType=Folder;
132  rc = metadata_setattr(folderuri, 'PublicType', 'Folder');
133  &dbg put 'NOTE- metadata_setattr (PublicType): ' rc=;
134  if ( rc ne 0 ) then do;
135  put "%str(ERR)OR: failed to set PublicType attribute for new folder: " walkpath;
136  stop;
137  end;
138 
139  * tag the new tree folder with the SAS 9.2 attribute UsageVersion=1000000;
140  rc = metadata_setattr(folderuri, 'UsageVersion', '1000000');
141  &dbg put 'NOTE- metadata_setattr (UsageVersion): ' rc=;
142  if ( rc ne 0 ) then do;
143  put "%str(ERR)OR: failed to set UsageVersion attribute for new folder: " walkpath;
144  stop;
145  end;
146 
147 
148  put 'NOTE- Sucessfully created new metadata folder: ' walkpath ;
149  parentFolderObjId=folderuri;
150  loop_next:
151  pathIndex + 1;
152  end;
153  &dbg put (_all_)(=);
154 run;
155 
156 %mend;
